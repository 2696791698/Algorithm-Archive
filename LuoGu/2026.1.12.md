## P1498 南蛮图腾

```c++
#include <algorithm>
#include <iostream>
#include <iomanip>
#include <string>
#include <sstream>
#include <cmath>
#include <vector>
#include <set>
#include <map>
#include <climits>
#include <numeric>
#include <deque>
#include <unordered_set>
#include <unordered_map>
#include <stack>
#include <queue>
#include <utility>
#include <variant>
#include <random>
#include <cstring>

using namespace std;
using ll = long long;
using ull = unsigned long long;

void dfs(int n, vector<string> &graph, string &padding) {
    if (n == 1) {
        for (auto x : graph) {
            cout << x << endl;
        }
        return;
    }
    int m = graph.size();
    for (int i = 0; i < m; i++) {
        graph.push_back(graph[i] + graph[i]);
    }
    for (int i = 0; i < m; i++) {
        graph[i] = padding + graph[i] + padding;
    }
    padding += padding;
    dfs(n - 1, graph, padding);
}

void solve() {
    vector<string> graph(2);
    graph[0] = " /\\ ";
    graph[1] = "/__\\";
    int n;
    cin >> n;
    string padding = "  ";
    dfs(n, graph, padding);
}

int main() {
    // ios::sync_with_stdio(false);
    // cin.tie(nullptr);

    if (0) {
        int n;
        cin >> n;
        while (n--) {
            solve();
        }
    } else {
        solve();
    }
}
```

## P1010 [NOIP 1998 普及组] 幂次方

```c++
#include <algorithm>
#include <iostream>
#include <iomanip>
#include <string>
#include <sstream>
#include <cmath>
#include <vector>
#include <set>
#include <map>
#include <climits>
#include <numeric>
#include <deque>
#include <unordered_set>
#include <unordered_map>
#include <stack>
#include <queue>
#include <utility>
#include <variant>
#include <random>
#include <cstring>

using namespace std;
using ll = long long;
using ull = unsigned long long;

string dfs(int n) {
    int tmp = n;
    vector<int> sup;
    int cnt = 0;
    while (tmp != 0) {
        if (tmp & 1) {
            sup.push_back(cnt);
        }
        cnt++;
        tmp >>= 1;
    }
    reverse(sup.begin(), sup.end());
    string res;
    for (auto x : sup) {
        if (x == 1) {
            res += "2+";
            continue;
        }
        if (x <= 2) {
            res += "2(";
            res += to_string(x);
            res += ")+";
            continue;
        }
        string s = dfs(x);
        res += "2(";
        res += s;
        res += ")+";
    }
    res.erase(res.size() - 1);
    return res;
}

void solve() {
    int n;
    cin >> n;
    string res;
    res = dfs(n);
    cout << res;
}

int main() {
    // ios::sync_with_stdio(false);
    // cin.tie(nullptr);

    if (0) {
        int n;
        cin >> n;
        while (n--) {
            solve();
        }
    } else {
        solve();
    }
}
```
