```c++
#include <algorithm>

using namespace std;
using ll = long long;
using ull = unsigned long long;

vector<int> deltaX = {0, 1, 0, -1};
vector<int> deltaY = {1, 0, -1, 0};

int dfs(int currX, int currY, int endX, int endY, int n, int m, vector<vector<int>> &graph) {
    if (currX == endX && currY == endY) {
        return 1;
    }
    int ans = 0;
    for (int i = 0; i < 4; i++) {
        int nextX = currX + deltaX[i];
        int nextY = currY + deltaY[i];
        if (nextX < 1 || nextX > n || nextY < 1 || nextY > m) {
            continue;
        }
        if (graph[nextX][nextY] == 1) {
            continue;
        }
        graph[currX][currY] = 1;
        ans += dfs(nextX, nextY, endX, endY, n, m, graph);
        graph[currX][currY] = 0;
    }
    return ans;
}

void solve() {
    int n, m, t;
    cin >> n >> m >> t;
    vector<vector<int>> graph(n + 1, vector<int>(m + 1, 0));
    int startX, startY, endX, endY;
    cin >> startX >> startY >> endX >> endY;
    for (int i = 1; i <= t; i++) {
        int x, y;
        cin >> x >> y;
        graph[x][y] = 1;
    }
    cout << dfs(startX, startY, endX, endY, n, m, graph);
}

int main() {
    // ios::sync_with_stdio(false);
    // cin.tie(nullptr);

    if (0) {
        int n;
        cin >> n;
        while (n--) {
            solve();
        }
    } else {
        solve();
    }
}
```
