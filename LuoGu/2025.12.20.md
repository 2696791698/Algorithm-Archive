## P4779 【模板】单源最短路径（标准版）

```c++
#include <algorithm>
#include <iostream>
#include <iomanip>
#include <string>
#include <sstream>
#include <cmath>
#include <vector>
#include <set>
#include <map>
#include <climits>
#include <numeric>
#include <deque>
#include <unordered_set>
#include <unordered_map>
#include <stack>
#include <queue>
#include <utility>
#include <variant>

using namespace std;
using ll = long long;
using ull = unsigned long long;

class cmp {
public:
    bool operator()(const pair<int, int> &a, const pair<int, int> &b) const {
        return a.second > b.second;
    }
};

class Edge {
public:
    int to, value;
    Edge(int t, int v) : to(t), value(v) {} // 列表初始化语法
};

void solve() {
    int n, m, s;
    cin >> n >> m >> s;
    vector<int> minDist(n + 1, INT_MAX);
    vector<bool> visited(n + 1, false);
    vector<vector<Edge>> edges(n + 1);
    priority_queue<pair<int, int>, vector<pair<int, int>>, cmp> pq;
    for (int i = 1; i <= m ;i++) {
        int from, to, value;
        cin >> from >> to >> value;
        edges[from].emplace_back(to, value);
    }
    minDist[s] = 0;
    pq.emplace(s, 0);
    while (pq.empty() == false) {
        auto cur = pq.top();
        pq.pop();
        if (visited[cur.first] == false) {
            visited[cur.first] = true;
        }
        for (auto x : edges[cur.first]) {
            if (visited[x.to] == false && minDist[x.to] > minDist[cur.first] + x.value) {
                minDist[x.to] = minDist[cur.first] + x.value;
                pq.emplace(x.to, minDist[x.to]);
            }
        }
    }
    for (int i = 1; i <= n; i++) {
        cout << minDist[i] << ' ';
    }
}

int main() {
    // ios::sync_with_stdio(false);
    // cin.tie(nullptr);

    solve();
}
```

## P3385 【模板】负环

```c++
#include <algorithm>
#include <iostream>
#include <iomanip>
#include <string>
#include <sstream>
#include <cmath>
#include <vector>
#include <set>
#include <map>
#include <climits>
#include <numeric>
#include <deque>
#include <unordered_set>
#include <unordered_map>
#include <stack>
#include <queue>
#include <utility>
#include <variant>

using namespace std;
using ll = long long;
using ull = unsigned long long;

void solve() {
    int n, m;
    cin >> n >> m;
    vector<vector<int>> edges;
    vector<int> minDist(n + 1, INT_MAX);
    for (int i = 1; i <= m; i++) {
        int from, to, value;
        cin >> from >> to >> value;
        if (value >= 0) {
            edges.push_back({from, to, value});
            edges.push_back({to, from, value});
        } else {
            edges.push_back({from, to, value});
        }
    }
    minDist[1] = 0;
    bool flag = false;
    for (int i = 1; i <= n; i++) { // 松弛n次
        if (i != n) {
            for (auto edge : edges) {
                int from = edge[0];
                int to = edge[1];
                int value = edge[2];
                if (minDist[from] == INT_MAX) {
                    continue;
                }
                if (minDist[to] > minDist[from] + value) {
                    minDist[to] = minDist[from] + value;
                }
            }
        } else { // 第n次用于检测负环
            for (auto edge : edges) {
                int from = edge[0];
                int to = edge[1];
                int value = edge[2];
                if (minDist[from] == INT_MAX) {
                    continue;
                }
                if (minDist[to] > minDist[from] + value) {
                    flag = true;
                    break;
                }
            }
        }
    }
    if (flag == false) {
        cout << "NO" << endl;
    } else {
        cout << "YES" << endl;
    }
}

int main() {
    // ios::sync_with_stdio(false);
    // cin.tie(nullptr);

    int t;
    cin >> t;
    while (t--) {
        solve();
    }
}
```
