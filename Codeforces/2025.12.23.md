using namespace std;
using ll = long long;
using ull = unsigned long long;

void dfs(vector<int> bits) {
    if (bits.size() == 1) {
        cout << bits[0] << " ";
        return;
    }
    vector<int> newBits;
    for (int i = 1; i < bits.size(); i += 2) {
        newBits.push_back(bits[i]);
    }
    dfs(newBits);
    for (int i = 0; i < bits.size(); i += 2) {
        cout << bits[i] << " ";
    }
}

void solve() {
    int n;
    cin >> n;
    vector<int> arrZero;
    vector<int> arrOne;
    for (int i = 0; i < (1 << n); i++) {
        if (i & 1) {
            arrOne.push_back(i);
        } else {
            arrZero.push_back(i);
        }
    }
    sort(arrZero.begin(), arrZero.end());
    dfs(arrOne);
    for (int x : arrZero) {
        cout << x << " ";
    }
    cout << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while (t--) {
        solve();
    }
}
